## Mac wtf-sync.sh 內容
（Mac 已貼，2026-06-03。Windows port 注意：`python3`→`python`、`date +%s`/`/tmp` 戳記改 Windows 等價、`find .git -name "*.lock" -delete` 改 PowerShell `Remove-Item`、路徑分隔符。）

```bash
#!/bin/bash
# WTF repo 開場自動同步：git pull + sync ~/.claude/
# 5 分鐘冷卻，避免同一 session 重複跑

STAMP=/tmp/wtf_sync_last
NOW=$(date +%s)
LAST=$(cat "$STAMP" 2>/dev/null || echo 0)
[ $((NOW - LAST)) -lt 300 ] && exit 0
echo $NOW > "$STAMP"

WTF="/Users/coma/Library/CloudStorage/GoogleDrive-coldjokenewbie@gmail.com/其他電腦/tachart_ihuy/Claude_cowork/projects/WTF_Under_Construction"

echo "[WTF-SYNC] 開始同步..."

# 清 Drive 殘留 lock 檔
find "$WTF/.git" -name "*.lock" -delete 2>/dev/null

PULL=$(cd "$WTF" && git pull 2>&1)
if [ $? -eq 0 ]; then
    echo "[WTF-SYNC] git pull: $PULL"
else
    echo "[WTF-SYNC] git pull 失敗: $PULL"
    exit 1
fi

SYNC=$(cd "$WTF" && python3 wtf-config/sync_config.py sync 2>&1)
if [ $? -eq 0 ]; then
    echo "[WTF-SYNC] sync 完成"
else
    echo "[WTF-SYNC] sync 失敗: $SYNC"
    exit 1
fi
```
